Common Queries

Find books with filters

const where: Prisma.BookWhereInput = {
  isPublished: true,
  ...(search && {
    OR: [
      { title: { contains: search, mode: 'insensitive' } },
      { author: { contains: search, mode: 'insensitive' } },
    ],
  }),
  ...(category && {
    categories: { some: { category: { slug: category } } },
  }),
};
